# SQLite

CREATE TABLE Person(
DogID REFERENCES Dog(DogID),
Password VARCHAR(60),
Email VARCHAR(255) PRIMARY KEY
);

CREATE TABLE Dog(
Weight INT,
DOB DATE,
Name CHARACTER(20),
DogID INT PRIMARY KEY
);

CREATE TABLE Poop(
DogID REFERENCES Dog(DogID),
Time DATETIME,
(DogID, Time) PRIMARY KEY
);

CREATE TABLE Pee(
DogID REFERENCES Dog(DogID),
Time DATETIME,
(DogID, Time) PRIMARY KEY
);

CREATE TABLE Feed(
DogID REFERENCES Dog(DogID),
Time DATETIME,
PRIMARY KEY(DogID, Time)
);

CREATE TABLE VetTrip(
DogID REFERENCES Dog(DogID),
Time DATETIME,
Treatment VARCHAR(255),
PRIMARY KEY(DogID, Time)
);
